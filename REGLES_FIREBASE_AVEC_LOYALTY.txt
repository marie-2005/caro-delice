rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Fonction helper pour vérifier si l'utilisateur est admin
    function isAdmin() {
      return request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.rôle == 'admin');
    }
    
    match /orders/{orderId} {
      // Admin en PREMIER pour qu'il voie TOUT
      allow read: if request.auth != null && (
        isAdmin() ||
        resource.data.customerId == request.auth.uid ||
        resource.data.customerId == null
      );
      allow create: if true;
      allow update, delete: if request.auth != null && isAdmin();
    }
    
    match /users/{userId} {
      allow read: if request.auth != null && 
        (request.auth.uid == userId || isAdmin());
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // ✨ NOUVELLES RÈGLES POUR LES POINTS FIDÉLITÉ
    match /loyalty/{userId} {
      // Les utilisateurs peuvent lire leur propre profil de fidélité
      allow read: if request.auth != null && request.auth.uid == userId;
      
      // Les admins peuvent lire tous les profils de fidélité
      allow read: if request.auth != null && isAdmin();
      
      // Les utilisateurs peuvent créer leur propre profil de fidélité
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Les utilisateurs peuvent mettre à jour leur propre profil de fidélité
      allow update: if request.auth != null && request.auth.uid == userId;
      
      // Les admins peuvent tout faire
      allow write: if request.auth != null && isAdmin();
    }
    
    // Règles pour l'historique des points fidélité
    match /loyalty/{userId}/history/{historyId} {
      // Les utilisateurs peuvent lire leur propre historique
      allow read: if request.auth != null && request.auth.uid == userId;
      
      // Les admins peuvent lire tout l'historique
      allow read: if request.auth != null && isAdmin();
      
      // Les utilisateurs peuvent créer leur historique
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Les admins peuvent tout faire
      allow write: if request.auth != null && isAdmin();
    }
  }
}
